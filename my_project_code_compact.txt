package ir.Ali.hollowknightme.controller;
import com.badlogic.gdx.math.Vector2;
import com.badlogic.gdx.physics.box2d.*;
import ir.Ali.hollowknightme.model.knight.Knight;
import ir.Ali.hollowknightme.model.map.GroundData;
import ir.Ali.hollowknightme.model.map.GroundType;
public class KnightContactListener implements ContactListener {
    private final Knight knight;
    private int groundContacts = 0;
    private int wallContacts = 0;
    private int ceilingContacts = 0;
    public KnightContactListener(Knight knight) {
        this.knight = knight;
    }
    @Override
    public void beginContact(Contact contact) {
        Body bodyA = contact.getFixtureA().getBody();
        Body bodyB = contact.getFixtureB().getBody();
        if (bodyA != knight.getBody() && bodyB != knight.getBody()) return;
        GroundData groundData = getGroundDataFromContact(contact);
        if (groundData != null) {
            System.out.println("DEBUG: Ground Contact Started! Type: " + groundData.getType());
            handleGroundContact(contact, groundData);
            return;
        }
        handleLegacyContact(contact);
    }
    @Override
    public void endContact(Contact contact) {
        Body bodyA = contact.getFixtureA().getBody();
        Body bodyB = contact.getFixtureB().getBody();
        if (bodyA != knight.getBody() && bodyB != knight.getBody()) return;
        GroundData groundData = getGroundDataFromContact(contact);
        if (groundData != null) {
            handleGroundEndContact(contact, groundData);
            return;
        }
        handleLegacyEndContact(contact);
    }
    private GroundData getGroundDataFromContact(Contact contact) {
        Object userDataA = contact.getFixtureA().getUserData();
        Object userDataB = contact.getFixtureB().getUserData();
        if (userDataA instanceof GroundData) return (GroundData) userDataA;
        if (userDataB instanceof GroundData) return (GroundData) userDataB;
        return null;
    }
    private void handleGroundContact(Contact contact, GroundData groundData) {
        WorldManifold worldManifold = contact.getWorldManifold();
        Vector2 normal = worldManifold.getNormal();
        if (normal.y > 0.5f && groundData.isTopActive()) {
            groundContacts++;
            knight.setGrounded(true);
            knight.setCanDoubleJump(true);
            return;
        }
        if (normal.y < -0.5f && groundData.isBottomActive()) {
            ceilingContacts++;
            knight.setOnCeiling(true);
        } else if (Math.abs(normal.x) > 0.5f) {
            if ((normal.x > 0 && groundData.isLeftActive()) ||
                (normal.x < 0 && groundData.isRightActive())) {
                wallContacts++;
                knight.setOnWall(true);
            }
        }
    }
    private void handleGroundEndContact(Contact contact, GroundData groundData) {
        groundContacts--;
        if (groundContacts <= 0) {
            groundContacts = 0;
            knight.setGrounded(false);
        }
    }
    private void handleLegacyContact(Contact contact) {
        Object userDataA = contact.getFixtureA().getUserData();
        Object userDataB = contact.getFixtureB().getUserData();
        String type = (userDataA instanceof String) ? (String) userDataA :
            (userDataB instanceof String) ? (String) userDataB : null;
        System.out.println("DEBUG: Legacy Contact with type: " + type);
        if (userDataA instanceof String) type = (String) userDataA;
        else if (userDataB instanceof String) type = (String) userDataB;
        if ("floor".equals(type)) {
            groundContacts++;
            knight.setGrounded(true);
            knight.setCanDoubleJump(true);
        } else if ("wall".equals(type)) {
            wallContacts++;
            knight.setOnWall(true);
        } else if ("ceiling".equals(type)) {
            ceilingContacts++;
            knight.setOnCeiling(true);
        } else {
            WorldManifold worldManifold = contact.getWorldManifold();
            Vector2 normal = worldManifold.getNormal();
            if (normal.y > 0.5f) {
                groundContacts++;
                knight.setGrounded(true);
                knight.setCanDoubleJump(true);
            } else if (Math.abs(normal.x) > 0.5f) {
                wallContacts++;
                knight.setOnWall(true);
            } else if (normal.y < -0.5f) {
                ceilingContacts++;
                knight.setOnCeiling(true);
            }
        }
    }
    private void handleLegacyEndContact(Contact contact) {
        Object userDataA = contact.getFixtureA().getUserData();
        Object userDataB = contact.getFixtureB().getUserData();
        String type = null;
        if (userDataA instanceof String) type = (String) userDataA;
        else if (userDataB instanceof String) type = (String) userDataB;
        if ("floor".equals(type)) {
            groundContacts--;
            if (groundContacts <= 0) {
                groundContacts = 0;
                knight.setGrounded(false);
            }
        } else if ("wall".equals(type)) {
            wallContacts--;
            if (wallContacts <= 0) {
                wallContacts = 0;
                knight.setOnWall(false);
            }
        } else if ("ceiling".equals(type)) {
            ceilingContacts--;
            if (ceilingContacts <= 0) {
                ceilingContacts = 0;
                knight.setOnCeiling(false);
            }
        }
    }
    @Override
    public void preSolve(Contact contact, Manifold oldManifold) {}
    @Override
    public void postSolve(Contact contact, ContactImpulse impulse) {}
}

package ir.Ali.hollowknightme.controller;
import com.badlogic.gdx.physics.box2d.Body;
import com.badlogic.gdx.math.Vector2;
import ir.Ali.hollowknightme.enums.knight.KnightState;
import ir.Ali.hollowknightme.enums.knight.MovementStatus;
import ir.Ali.hollowknightme.manager.game.ControlManager;
import ir.Ali.hollowknightme.manager.game.GameStateManager;
import ir.Ali.hollowknightme.model.knight.Knight;
import ir.Ali.hollowknightme.view.knight.KnightAnimator;
public class KnightController {
    private final Knight knight;
    private final Body body;
    private static final float MAX_VELOCITY = 1.5f;
    private static final float JUMP_IMPULSE = 6f;
    private static final float JUMP_FORCE = 0.9f;
    private static final float MAX_JUMP_VELOCITY = 10f;
    private static final float JUMP_CUTOFF_FACTOR = 0.7f;
    private static final float DASH_IMPULSE = 8f;
    private static final float GRAVITY_SCALE = 1f;
    private float dashTimer = 0f;
    private float attackTimer = 0f;
    private float focusTimer = 0f;
    private float landingTimer = 0f;
    private boolean isJumping = false;
    private float jumpHoldTime = 0f;
    private boolean previousGrounded = false;
    public KnightController(Knight knight) {
        this.knight = knight;
        this.body = knight.getBody();
        this.previousGrounded = knight.isGrounded();
    }
    public void update(float deltaTime) {
        updateTimers(deltaTime);
        handleInput(deltaTime);
        handlePhysics();
        updateState();
        checkLanding();
    }
    private void updateTimers(float deltaTime) {
        if (dashTimer > 0) dashTimer -= deltaTime;
        if (attackTimer > 0) attackTimer -= deltaTime;
        if (focusTimer > 0) focusTimer -= deltaTime;
        if (landingTimer > 0) landingTimer -= deltaTime;
    }
    private void handleInput(float deltaTime) {
        ControlManager cm = ControlManager.getInstance();
        if (dashTimer > 0 || attackTimer > 0 || focusTimer > 0 ||
            knight.getCurrentState() == KnightState.TRANSITION) {
            return;
        }
        if (knight.isLocked()) {
            return;
        }
        Vector2 velocity = body.getLinearVelocity();
        float targetSpeedX = 0f;
        if (cm.isActionPressed(MovementStatus.MOVE_RIGHT)) {
            targetSpeedX = MAX_VELOCITY;
            knight.setFacingRight(true);
        } else if (cm.isActionPressed(MovementStatus.MOVE_LEFT)) {
            targetSpeedX = -MAX_VELOCITY;
            knight.setFacingRight(false);
        }
        body.setLinearVelocity(targetSpeedX, velocity.y);
        boolean jumpPressed = cm.isActionPressed(MovementStatus.JUMP);
        boolean jumpJustPressed = cm.isActionJustPressed(MovementStatus.JUMP);
        if (jumpJustPressed) {
            if (knight.isGrounded()) {
                doJump(JUMP_IMPULSE);
                knight.setGrounded(false);
                knight.setCanDoubleJump(true);
                isJumping = true;
                jumpHoldTime = 0f;
            } else if (knight.isOnWall() && !knight.isGrounded()) {
                float wallJumpImpulse = JUMP_IMPULSE * 1.2f;
                float dir = knight.isFacingRight() ? -1 : 1;
                body.setLinearVelocity(dir * MAX_VELOCITY * 0.8f, wallJumpImpulse);
                knight.setOnWall(false);
                knight.setCanDoubleJump(true);
                isJumping = true;
                jumpHoldTime = 0f;
            } else if (knight.isCanDoubleJump()) {
                float doubleJumpImpulse = JUMP_IMPULSE * 0.9f;
                body.setLinearVelocity(velocity.x, 0);
                doJump(doubleJumpImpulse);
                knight.setCanDoubleJump(false);
                isJumping = true;
                jumpHoldTime = 0f;
            }
        }
        if (jumpPressed && isJumping && velocity.y < MAX_JUMP_VELOCITY) {
            body.applyForceToCenter(0, JUMP_FORCE, true);
            jumpHoldTime += deltaTime;
        }
        if (!jumpPressed && isJumping && velocity.y > 0) {
            body.setLinearVelocity(velocity.x, velocity.y * JUMP_CUTOFF_FACTOR);
            isJumping = false;
        }
        if (cm.isActionJustPressed(MovementStatus.NAIL_ATTACK)) {
            attackTimer = KnightAnimator.getAnimationDuration(KnightState.SLASH);
        }
        if (cm.isActionJustPressed(MovementStatus.DASH) && dashTimer <= 0) {
            dashTimer = KnightAnimator.getAnimationDuration(KnightState.DASH);
            float direction = knight.isFacingRight() ? 1 : -1;
            body.setLinearVelocity(direction * DASH_IMPULSE, 0);
            body.setGravityScale(0);
            knight.setDashing(true);
        }
        if (cm.isActionPressed(MovementStatus.FOCUS)) {
            focusTimer = 1.5f;
        }
    }
    private void doJump(float impulse) {
        body.applyLinearImpulse(new Vector2(0, impulse), body.getWorldCenter(), true);
    }
    private void handlePhysics() {
        if (dashTimer > 0) {
            body.setGravityScale(0);
        } else {
            body.setGravityScale(GRAVITY_SCALE);
            if (knight.isDashing()) {
                knight.setDashing(false);
            }
        }
        Vector2 vel = body.getLinearVelocity();
        if (Math.abs(vel.x) > MAX_VELOCITY && dashTimer <= 0) {
            vel.x = Math.signum(vel.x) * MAX_VELOCITY;
            body.setLinearVelocity(vel);
        }
        if (knight.isOnWall() && Math.abs(vel.x) < 0.1f) {
            knight.setOnWall(false);
        }
        if (knight.isOnCeiling() && vel.y > 0) {
            body.setLinearVelocity(vel.x, 0);
            knight.setOnCeiling(false);
        }
    }
    private void checkLanding() {
        boolean currentGrounded = knight.isGrounded();
        if (!previousGrounded && currentGrounded) {
            landingTimer = 0.15f;
        }
        previousGrounded = currentGrounded;
    }
    private void updateState() {
        if (knight.getCurrentState() == KnightState.TRANSITION) {
            return;
        }
        Vector2 vel = body.getLinearVelocity();
        if (focusTimer > 0) {
            knight.setCurrentState(KnightState.FOCUS);
            return;
        }
        if (dashTimer > 0) {
            knight.setCurrentState(KnightState.DASH);
            return;
        }
        if (attackTimer > 0) {
            if (!knight.isGrounded() && vel.y < -0.1f) {
                knight.setCurrentState(KnightState.DOWN_SLASH);
            } else {
                knight.setCurrentState(KnightState.SLASH);
            }
            return;
        }
        if (landingTimer > 0) {
            knight.setCurrentState(KnightState.LANDING);
            return;
        }
        if (knight.isOnWall() && !knight.isGrounded() && Math.abs(vel.y) > 0.1f) {
            knight.setCurrentState(KnightState.WALL_SLIDE);
            return;
        }
        if (!knight.isGrounded() && !knight.isCanDoubleJump() && vel.y > 0.5f) {
            knight.setCurrentState(KnightState.DOUBLE_JUMP);
            return;
        }
        if (!knight.isGrounded() && Math.abs(vel.y) > 0.1f) {
            knight.setCurrentState(KnightState.AIRBORNE);
            return;
        }
        if (Math.abs(vel.x) > 0.1f) {
            knight.setCurrentState(KnightState.RUN);
        } else {
            knight.setCurrentState(KnightState.IDLE);
        }
    }
    public void setLandingTimer(float duration) {
        this.landingTimer = duration;
    }
    public void resetJumpState() {
        isJumping = false;
        jumpHoldTime = 0f;
    }
    public void respawn() {
        float[] safePos = GameStateManager.getInstance().getSafePosition();
        if (safePos != null) {
            knight.respawn(safePos[0], safePos[1]);
            isJumping = false;
            jumpHoldTime = 0f;
            previousGrounded = false;
            dashTimer = 0f;
            attackTimer = 0f;
            focusTimer = 0f;
            landingTimer = 0f;
        }
    }
}

package ir.Ali.hollowknightme.manager;
import com.badlogic.gdx.physics.box2d.Contact;
import com.badlogic.gdx.physics.box2d.ContactImpulse;
import com.badlogic.gdx.physics.box2d.ContactListener;
import com.badlogic.gdx.physics.box2d.Manifold;
import java.util.ArrayList;
import java.util.List;
public class CompositeContactListener implements ContactListener {
    private final List<ContactListener> listeners = new ArrayList<>();
    public void addListener(ContactListener listener) {
        if (listener != null) {
            listeners.add(listener);
        }
    }
    @Override
    public void beginContact(Contact contact) {
        for (ContactListener l : listeners) {
            l.beginContact(contact);
        }
    }
    @Override
    public void endContact(Contact contact) {
        for (ContactListener l : listeners) {
            l.endContact(contact);
        }
    }
    @Override
    public void preSolve(Contact contact, Manifold oldManifold) {
        for (ContactListener l : listeners) {
            l.preSolve(contact, oldManifold);
        }
    }
    @Override
    public void postSolve(Contact contact, ContactImpulse impulse) {
        for (ContactListener l : listeners) {
            l.postSolve(contact, impulse);
        }
    }
}

package ir.Ali.hollowknightme.manager;
import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input.Keys;
import com.badlogic.gdx.utils.ObjectMap;
import ir.Ali.hollowknightme.enums.knight.MovementStatus;
public class ControlManager {
    private static volatile ControlManager instance;
    private final ObjectMap<MovementStatus, Integer> keybinds;
    private ControlManager() {
        this.keybinds = new ObjectMap<>();
        loadDefaultKeybinds();
    }
    public static ControlManager getInstance() {
        if (instance == null) {
            synchronized (ControlManager.class) {
                if (instance == null) {
                    instance = new ControlManager();
                }
            }
        }
        return instance;
    }
    private void loadDefaultKeybinds() {
        keybinds.put(MovementStatus.MOVE_LEFT, Keys.LEFT);
        keybinds.put(MovementStatus.MOVE_RIGHT, Keys.RIGHT);
        keybinds.put(MovementStatus.JUMP, Keys.UP);
        keybinds.put(MovementStatus.LOOK_UP, Keys.W);
        keybinds.put(MovementStatus.LOOK_DOWN, Keys.S);
        keybinds.put(MovementStatus.DASH, Keys.SHIFT_LEFT);
        keybinds.put(MovementStatus.NAIL_ATTACK, Keys.K);
    }
    public boolean isActionPressed(MovementStatus action) {
        Integer keyCode = keybinds.get(action);
        return keyCode != null && Gdx.input.isKeyPressed(keyCode);
    }
    public boolean isActionJustPressed(MovementStatus action) {
        Integer keyCode = keybinds.get(action);
        return keyCode != null && Gdx.input.isKeyJustPressed(keyCode);
    }

public class PhysicsBuilder {
    private static final float PPM = 100f;
    public static List<Body> buildPhysics(World world, MapData mapData) {
        List<Body> bodies = new ArrayList<>();
        if (mapData == null) return bodies;
        for (GroundData ground : mapData.getGrounds()) {
            Body body = createGroundBody(world, ground);
            if (body != null) bodies.add(body);
        }
        for (ir.Ali.hollowknightme.model.map.TransitionData transition : mapData.getTransitions()) {
            Body body = createTransitionBody(world, transition);
            if (body != null) bodies.add(body);
        }
        return bodies;
    }
    private static Body createTransitionBody(World world, ir.Ali.hollowknightme.model.map.TransitionData transition) {
        Rectangle rect = transition.getRectangle();
        float centerX = (rect.x + rect.width / 2f) / PPM;
        float centerY = (rect.y + rect.height / 2f) / PPM;
        float halfWidth = (rect.width / 2f) / PPM;
        float halfHeight = (rect.height / 2f) / PPM;
        BodyDef bodyDef = new BodyDef();
        bodyDef.type = BodyDef.BodyType.StaticBody;
        bodyDef.position.set(centerX, centerY);
        Body body = world.createBody(bodyDef);
        PolygonShape shape = new PolygonShape();
        shape.setAsBox(halfWidth, halfHeight);
        FixtureDef fixtureDef = new FixtureDef();
        fixtureDef.shape = shape;
        fixtureDef.isSensor = true;
        Fixture fixture = body.createFixture(fixtureDef);
        fixture.setUserData(transition);
        shape.dispose();
        body.setUserData("transition");
        return body;
    }
    private static Body createGroundBody(World world, GroundData ground) {
        Rectangle rect = ground.getRectangle();
        GroundType type = ground.getType();
        float centerX = (rect.x + rect.width / 2f) / PPM;
        float centerY = (rect.y + rect.height / 2f) / PPM;
        float halfWidth = (rect.width / 2f) / PPM;
        float halfHeight = (rect.height / 2f) / PPM;
        BodyDef bodyDef = new BodyDef();
        bodyDef.type = BodyDef.BodyType.StaticBody;
        bodyDef.position.set(centerX, centerY);
        Body body = world.createBody(bodyDef);
        PolygonShape shape = new PolygonShape();
        shape.setAsBox(halfWidth, halfHeight);
        FixtureDef fixtureDef = new FixtureDef();
        fixtureDef.shape = shape;

        fixtureDef.restitution = 0f;

package ir.Ali.hollowknightme.manager;
import com.badlogic.gdx.maps.MapObject;
import com.badlogic.gdx.maps.MapObjects;
import com.badlogic.gdx.maps.objects.PointMapObject;
import com.badlogic.gdx.maps.objects.RectangleMapObject;
import com.badlogic.gdx.math.Rectangle;
import com.badlogic.gdx.math.Vector2;
import ir.Ali.hollowknightme.model.map.*;
public class TiledMapLoader {
    public static MapData loadMapData(MapObjects collisionObjects, MapObjects entityObjects) {
        MapData mapData = new MapData();
        if (collisionObjects != null) {
            for (MapObject object : collisionObjects) {
                if (!(object instanceof RectangleMapObject)) continue;
                Rectangle rect = ((RectangleMapObject) object).getRectangle();
                String className = getClassName(object).trim();
                if (isGroundClass(className)) {
                    GroundData ground = parseGround(object, rect, className);
                    if (ground != null) mapData.addGround(ground);
                }
            }
        }
        if (entityObjects != null) {
            for (MapObject object : entityObjects) {
                String className = getClassName(object).trim();
                if (className.equalsIgnoreCase("spawn")) {
                    SpawnData spawn = parseSpawn(object);
                    if (spawn != null) {
                        mapData.addSpawn(spawn);
                    }
                } else if (className.equalsIgnoreCase("transition") ||
                    className.equalsIgnoreCase("input") ||
                    className.equalsIgnoreCase("output")) {
                    TransitionData transition = parseTransition(object);
                    if (transition != null) {
                        mapData.addTransition(transition);
                    } else {
                        System.err.println("ERROR: Transition parsed as null for: " + object.getName());
                    }
                } else if (className.equalsIgnoreCase("range")) {
                    RangeData range = parseRange(object);
                    if (range != null) {
                        mapData.addRange(range);
                    }
                } else {
                    System.err.println("WARNING: Unknown entity object ignored: " + object.getName() + " (Class identified as: " + className + ")");
                }
            }
        }
        return mapData;
    }
    private static boolean isGroundClass(String className) {
        if (className == null) return false;
        String lower = className.toLowerCase();
        return lower.equals("ground") ||
            lower.equals("floor") ||
            lower.equals("wall") ||
            lower.equals("platform") ||
            lower.equals("thorn");
    }
    private static String getClassName(MapObject object) {
        Object value = object.getProperties().get("class");
        if (value == null) value = object.getProperties().get("type");
        if (value == null) value = object.getName();
        return value == null ? "" : value.toString();
    }
    private static String getStringProperty(MapObject object, String key, String defaultValue) {
        Object value = object.getProperties().get(key);
        return value == null ? defaultValue : value.toString();
    }
    private static boolean getBooleanProperty(MapObject object, String key, boolean defaultValue) {
        Object value = object.getProperties().get(key);
        if (value == null) return defaultValue;
        if (value instanceof Boolean) return (Boolean) value;
        return Boolean.parseBoolean(value.toString());
    }
    private static float getFloatProperty(MapObject object, String key, float defaultValue) {
        Object value = object.getProperties().get(key);
        if (value == null) return defaultValue;
        if (value instanceof Float) return (Float) value;
        if (value instanceof Integer) return ((Integer) value).floatValue();
        if (value instanceof Double) return ((Double) value).floatValue();
        try {
            return Float.parseFloat(value.toString());
        } catch (Exception e) {
            return defaultValue;
        }
    }
    private static GroundData parseGround(MapObject object, Rectangle rect, String className) {
        try {
            String typeStr = getStringProperty(object, "type", null);
            if (typeStr == null) {
                typeStr = className;
            }
            GroundType type = GroundType.valueOf(typeStr.toUpperCase());
            boolean left = getBooleanProperty(object, "isActiveLeft", false);
            boolean right = getBooleanProperty(object, "isActiveRight", false);
            boolean top = getBooleanProperty(object, "isActiveTop", false);
            boolean bottom = getBooleanProperty(object, "isActiveBottom", false);
            if ((type == GroundType.FLOOR || type == GroundType.PLATFORM) &&
                !object.getProperties().containsKey("isActiveTop")) {
                top = true;
            }
            return new GroundData(rect, type, left, right, top, bottom);
        } catch (Exception e) {
            System.err.println("Ground Parse Error: " + e.getMessage());
            e.printStackTrace();
            return null;
        }
    }
    private static SpawnData parseSpawn(MapObject object) {
        try {
            Vector2 position;
            if (object instanceof RectangleMapObject) {
                Rectangle rect = ((RectangleMapObject) object).getRectangle();
                position = new Vector2(rect.x + rect.width / 2f, rect.y + rect.height / 2f);
            } else if (object instanceof PointMapObject) {
                PointMapObject point = (PointMapObject) object;
                position = new Vector2(point.getPoint().x, point.getPoint().y);
            } else {
                float x = getFloatProperty(object, "x", 0);
                float y = getFloatProperty(object, "y", 0);
                position = new Vector2(x, y);
            }
            String spawnTypeStr = getStringProperty(object, "spawnType", "PLAYER");
            SpawnType spawnType = SpawnType.valueOf(spawnTypeStr.toUpperCase());
            boolean facingRight = getBooleanProperty(object, "facingRight", true);
            return new SpawnData(position, spawnType, facingRight);
        } catch (Exception e) {
            System.err.println("Spawn Parse Error: " + e.getMessage());
            e.printStackTrace();
            return null;
        }
    }
    private static TransitionData parseTransition(MapObject object) {
        try {
            if (!(object instanceof RectangleMapObject)) return null;
            Rectangle rect = ((RectangleMapObject) object).getRectangle();
            String typeStr = getStringProperty(object, "type", "INPUT");
            TransitionType type = TransitionType.valueOf(typeStr.toUpperCase());
            return new TransitionData(rect, type);
        } catch (Exception e) {
            System.err.println("Transition Parse Error: " + e.getMessage());
            e.printStackTrace();
            return null;
        }
    }
    private static RangeData parseRange(MapObject object) {
        try {
            if (!(object instanceof RectangleMapObject)) return null;
            Rectangle rect = ((RectangleMapObject) object).getRectangle();
            String typeStr = getStringProperty(object, "rangeType", "PATROL");
            RangeType type = RangeType.valueOf(typeStr.toUpperCase());
            return new RangeData(rect, type);
        } catch (Exception e) {
            System.err.println("Range Parse Error: " + e.getMessage());
            e.printStackTrace();
            return null;
        }
    }
}

public class TransitionManager implements ContactListener {
    private final Knight knight;
    private final MapManager mapManager;
    private final World world;
    private final PhysicsBuilder physicsBuilder;
    private boolean isTransitioning = false;
    private List<Body> worldBodies = new ArrayList<>();
    private boolean pendingTransition = false;
    private Room targetRoomForTransition = null;
    public TransitionManager(Knight knight, MapManager mapManager, World world) {
        this.knight = knight;
        this.mapManager = mapManager;
        this.world = world;
        this.physicsBuilder = new PhysicsBuilder();
    }
    public void update() {
        if (pendingTransition && targetRoomForTransition != null) {
            performTransition(targetRoomForTransition);
            pendingTransition = false;
            targetRoomForTransition = null;
        }
    }
    @Override
    public void beginContact(Contact contact) {
        if (isTransitioning || pendingTransition) return;
        Fixture fixtureA = contact.getFixtureA();
        Fixture fixtureB = contact.getFixtureB();
        if (fixtureA.getBody() != knight.getBody() && fixtureB.getBody() != knight.getBody()) return;
        TransitionData transition = getTransitionData(fixtureA, fixtureB);
        if (transition == null) return;
        if (transition.isInput()) {
            targetRoomForTransition = GameStateManager.getInstance().getPreviousRoom();
            if (targetRoomForTransition != null) {
                pendingTransition = true;
            } else {
                System.err.println("TransitionManager: No previous room available!");
            }
        } else if (transition.isOutput()) {
            targetRoomForTransition = GameStateManager.getInstance().getNextRoom();
            if (targetRoomForTransition != null) {
                pendingTransition = true;
            } else {
                System.err.println("TransitionManager: No next room available!");
            }
        }
    }
    @Override
    public void endContact(Contact contact) {
    }
    @Override
    public void preSolve(Contact contact, Manifold oldManifold) {}
    @Override
    public void postSolve(Contact contact, ContactImpulse impulse) {}
    private void performTransition(Room targetRoom) {
        isTransitioning = true;
        knight.setCurrentState(KnightState.TRANSITION);
        knight.setLocked(true);
        String targetMap = targetRoom.getMapPath();
        if (targetMap == null || targetMap.isEmpty()) {
            System.err.println("TransitionManager: targetMap is null or empty!");
            unlockKnight();
            isTransitioning = false;
            return;
        }
        saveCurrentRoomState();
        mapManager.loadMap(targetMap);
        restoreRoomState(targetRoom.getState());
        rebuildWorld();
        placeKnightInNewMap();
        GameStateManager gsm = GameStateManager.getInstance();
        for (int i = 0; i < gsm.getCurrentEnvironment().getRooms().size(); i++) {
            if (gsm.getCurrentEnvironment().getRooms().get(i).getMapPath().equals(targetMap)) {
                gsm.moveToRoom(i);
                break;
            }
        }
        unlockKnight();
        knight.setCurrentState(KnightState.IDLE);
        isTransitioning = false;
    }
    private void saveCurrentRoomState() {
        RoomState state = GameStateManager.getInstance().getCurrentRoomState();
        if (state == null) return;
    }
    private void restoreRoomState(RoomState state) {
        if (state == null) return;
    }
    private void rebuildWorld() {
        for (Body body : worldBodies) {
            world.destroyBody(body);
        }
        worldBodies.clear();
        MapData mapData = mapManager.getMapData();
        if (mapData != null) {
            List<Body> newBodies = physicsBuilder.buildPhysics(world, mapData);
            worldBodies.addAll(newBodies);
        }
    }
    private void placeKnightInNewMap() {
        MapData mapData = mapManager.getMapData();
        if (mapData == null) {
            System.err.println("TransitionManager: MapData is null!");
            return;
        }
        Vector2 spawnPosition = null;
        for (SpawnData spawn : mapData.getSpawns()) {
            if (spawn.getType() == SpawnType.PLAYER) {
                spawnPosition = spawn.getPosition().cpy().scl(1f / MapManager.PPM);
                break;
            }
        }
        if (spawnPosition == null) {
            System.err.println("TransitionManager: No player spawn found in new map! Placing at (0,0).");
            spawnPosition = new Vector2(0, 0);
        }
        GameStateManager.getInstance().updateSafePosition(spawnPosition.x, spawnPosition.y);
        Body knightBody = knight.getBody();
        knightBody.setTransform(spawnPosition, 0);
        knightBody.setLinearVelocity(0, 0);
    }
    private void unlockKnight() {
        knight.setLocked(false);
    }
    private TransitionData getTransitionData(Fixture fixtureA, Fixture fixtureB) {
        Object dataA = fixtureA.getUserData();
        Object dataB = fixtureB.getUserData();
        if (dataA instanceof TransitionData) return (TransitionData) dataA;
        if (dataB instanceof TransitionData) return (TransitionData) dataB;
        return null;
    }
    public void setWorldBodies(List<Body> bodies) {
        this.worldBodies = bodies;
    }

        atlas = new TextureAtlas(Gdx.files.internal("assets/skin.atlas"));
        skin = new Skin(Gdx.files.internal("assets/skin.json"), atlas);

        if (skin == null) {
            load();

        return skin;

        if (atlas == null) {
            load();

        return atlas;

        if (skin != null) {
            skin.dispose();
            skin = null;
        }
        if (atlas != null) {
            atlas.dispose();
            atlas = null;
        }
    }

    private final Label mainLabel;
    private final Label notificationLabel;
    private final Skin skin = UIManager.getSkin();
    private boolean checked = false;
    public CheckBoxDi(String text, String onStatus) {
        checkboxImage = new Image(skin, "box_off");
        mainLabel = new Label(text, skin, "Text");
        Label.LabelStyle smallStyle;
        if (skin.has("default", Label.LabelStyle.class)) {
            smallStyle = new Label.LabelStyle(skin.get("default", Label.LabelStyle.class));
        } else {
            smallStyle = new Label.LabelStyle();
            smallStyle.font = skin.getFont("default");
            smallStyle.fontColor = Color.WHITE;
        }
        notificationLabel = new Label(" (" + onStatus + ") ", smallStyle);
        notificationLabel.setVisible(false);
        this.left();
        this.add(mainLabel).left();
        this.add(checkboxImage).left().padLeft(15f);
        this.add(notificationLabel).left();
        this.pack();
        this.addListener(new ClickListener() {
            @Override
            public void clicked(InputEvent event, float x, float y) {
                checked = !checked;
                notificationLabel.clearActions();
                if (checked) {
                    checkboxImage.setDrawable(skin, "box_on");
                    notificationLabel.setVisible(true);
                    notificationLabel.setColor(1, 1, 1, 0);
                    notificationLabel.setX(checkboxImage.getX() + checkboxImage.getWidth());
                    notificationLabel.addAction(Actions.sequence(
                        Actions.parallel(
                            Actions.fadeIn(0.25f),
                            Actions.moveBy(20f, 0f, 0.25f)
                        ),
                        Actions.delay(1.8f),
                        Actions.parallel(
                            Actions.fadeOut(0.25f),
                            Actions.moveBy(-20f, 0f, 0.25f)
                        ),
                        Actions.run(new Runnable() {
                            @Override
                            public void run() {
                                notificationLabel.setVisible(false);
                            }
                        })
                    ));
                } else {
                    checkboxImage.setDrawable(skin, "box_off");
                    notificationLabel.setVisible(false);
                }
            }
        });
    }
    public boolean isChecked() {
        return checked;
    }

package ir.Ali.hollowknightme.model;
import java.util.ArrayList;
public class Environment {
    private String name;
    private ArrayList<Room> rooms;
    private int currentRoomIndex;
    public Environment(String name) {
        this.name = name;
        this.rooms = new ArrayList<>();
        this.currentRoomIndex = 0;
    }
    public void addRoom(Room room) {
        rooms.add(room);
    }
    public Room getCurrentRoom() {
        if (currentRoomIndex >= 0 && currentRoomIndex < rooms.size()) {
            return rooms.get(currentRoomIndex);
        }
        return null;
    }
    public Room getNextRoom() {
        int nextIndex = currentRoomIndex + 1;
        if (nextIndex < rooms.size()) {
            return rooms.get(nextIndex);
        }
        return null;
    }
    public Room getPreviousRoom() {
        int prevIndex = currentRoomIndex - 1;
        if (prevIndex >= 0) {
            return rooms.get(prevIndex);
        }
        return null;
    }
    public void moveToNextRoom() {
        if (currentRoomIndex + 1 < rooms.size()) {
            currentRoomIndex++;
        }
    }
    public void moveToPreviousRoom() {
        if (currentRoomIndex - 1 >= 0) {
            currentRoomIndex--;
        }
    }
    public void moveToRoom(int index) {
        if (index >= 0 && index < rooms.size()) {
            currentRoomIndex = index;
        }
    }
    public int getCurrentRoomIndex() { return currentRoomIndex; }
    public ArrayList<Room> getRooms() { return rooms; }
    public String getName() { return name; }
}

        isBossDefeated = bossDefeated;
    }

        return isVisited;
    }

        isVisited = visited;
    }
    public float getLastSafeX() {
        return lastSafeX;
    }
    public void setLastSafeX(float lastSafeX) {
        this.lastSafeX = lastSafeX;
    }
    public float getLastSafeY() {
        return lastSafeY;
    }
    public void setLastSafeY(float lastSafeY) {
        this.lastSafeY = lastSafeY;
    }
    public void addAliveEnemy(Enemy enemy) {
        aliveEnemies.add(enemy);
    }
    public void moveEnemyToDead(Enemy enemy) {
        aliveEnemies.removeValue(enemy, true);
        deadEnemies.add(enemy);
    }
    public boolean isObjectDestroyed(String objectId) {
        return destroyedObjects.contains(objectId, false);
    }
    public void destroyObject(String objectId) {
        if (!destroyedObjects.contains(objectId, false)) {
            destroyedObjects.add(objectId);
        }
    }
    public void reset() {
        aliveEnemies.clear();
        deadEnemies.clear();
        destroyedObjects.clear();
        isBossDefeated = false;
        isVisited = false;
    }
}

package ir.Ali.hollowknightme.view;
import com.badlogic.gdx.graphics.g2d.Animation;
import com.badlogic.gdx.graphics.g2d.TextureAtlas;
import com.badlogic.gdx.graphics.g2d.TextureRegion;
import com.badlogic.gdx.utils.Array;
import ir.Ali.hollowknightme.enums.knight.KnightState;
import ir.Ali.hollowknightme.model.knight.Knight;
import java.util.EnumMap;
public class KnightAnimator {
    private final Knight knight;
    private static final EnumMap<KnightState, Animation<TextureRegion>> animations =
        new EnumMap<>(KnightState.class);
    private float stateTime = 0f;
    private KnightState previousState = KnightState.IDLE;
    public KnightAnimator(Knight knight, TextureAtlas atlas) {
        this.knight = knight;
        initAnimations(atlas);
    }
    private void initAnimations(TextureAtlas atlas) {
        for (KnightState state : KnightState.values()) {
            String atlasRegionName = getAtlasRegionName(state);
            Array<TextureAtlas.AtlasRegion> regions = atlas.findRegions(atlasRegionName);
            if (regions.size == 0) {
                state = KnightState.IDLE;
                atlasRegionName = getAtlasRegionName(state);
                regions = atlas.findRegions(atlasRegionName);
            }
            float frameDuration = getFrameDurationForState(state);
            Animation.PlayMode playMode = getPlayModeForState(state);
            Animation<TextureRegion> animation = new Animation<>(frameDuration, regions, playMode);
            animations.put(state, animation);
        }
    }
    private String getAtlasRegionName(KnightState state) {
        switch (state) {
            case DOUBLE_JUMP: return "Double Jump";
            case DOWN_SLASH: return "DownSlash";
            case FIREBALL_CAST: return "Fireball Cast";
            case FOCUS_GET: return "Focus Get";
            case FOCUS_START: return "Focus Start";
            case FOCUS_END: return "Focus End";
            case IDLE_HURT: return "Idle Hurt";
            case LOOK_DOWN: return "LookDown";
            case LOOK_UP: return "LookUp";
            case SLASH_ALT: return "SlashAlt";
            case UP_SLASH: return "UpSlash";
            case WALL_SLIDE: return "Wall Slide";
            case WALL_JUMP: return "Walljump";
            case TRANSITION: return "run";
            default:
                String name = state.name().toLowerCase();
                return name.substring(0, 1).toUpperCase() + name.substring(1);
        }
    }
    private float getFrameDurationForState(KnightState state) {
        switch (state) {
            case DASH: return 0.04f;
            case RUN: return 0.1f;
            case SLASH: return 0.9f;
            case SLASH_ALT: return 0.05f;
            case DEATH: return 0.18f;
            default: return 0.12f;
        }
    }
    private Animation.PlayMode getPlayModeForState(KnightState state) {
        switch (state) {
            case IDLE:
            case RUN:
            case AIRBORNE:
            case WALL_SLIDE:
            case FOCUS:
                return Animation.PlayMode.LOOP;
            default:
                return Animation.PlayMode.NORMAL;
        }
    }
    public TextureRegion getFrame(float deltaTime, KnightState currentState) {
        if (currentState != previousState) {
            stateTime = 0f;
        } else {
            stateTime += deltaTime;
        }
        previousState = currentState;
        Animation<TextureRegion> currentAnimation = animations.get(currentState);
        if (currentAnimation == null) return null;
        TextureRegion frame = currentAnimation.getKeyFrame(stateTime);
        if (knight.isFacingRight() && !frame.isFlipX()) {
            frame.flip(true, false);
        } else if (!knight.isFacingRight() && frame.isFlipX()) {
            frame.flip(true, false);
        }
        return frame;
    }
    public static float getAnimationDuration(KnightState state) {
        Animation<TextureRegion> anim = animations.get(state);
        if (anim == null) return 0.3f;
        return anim.getAnimationDuration();
    }
}

    }

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.GL20;
import com.badlogic.gdx.graphics.OrthographicCamera;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.graphics.g2d.TextureAtlas;

import java.util.List;
public class GameScreen extends BaseScreen {
    private final OrthographicCamera camera;
    private final Viewport viewport;
    private final SpriteBatch batch;
    private final MapManager mapManager;
    private final World world;
    private final Box2DDebugRenderer box2DDebugRenderer;
    private final Knight knight;
    private final KnightController knightController;
    private final KnightAnimator knightAnimator;
    private final TextureAtlas knightAtlas;
    private TransitionManager transitionManager;
    private CompositeContactListener compositeContactListener;
    private List<com.badlogic.gdx.physics.box2d.Body> worldBodies;
    public GameScreen(String mapPath) {
        super();
        camera = new OrthographicCamera();
        viewport = new FitViewport(1920f / Knight.PPM, 1080f / Knight.PPM, camera);
        batch = new SpriteBatch();
        world = new World(new Vector2(0, -10f), true);
        box2DDebugRenderer = new Box2DDebugRenderer();
        mapManager = new MapManager();
        mapManager.loadMap(mapPath);
        MapData mapData = mapManager.getMapData();
        Environment environment = new Environment("Default");
        Room room = new Room(mapPath, mapData);
        environment.addRoom(room);
        MapManager secondManager = new MapManager();
        secondManager.loadMap("room2.tmx");
        Room room2 = new Room("room2.tmx", secondManager.getMapData());
        environment.addRoom(room2);
        GameStateManager.getInstance().setCurrentEnvironment(environment);
        worldBodies = PhysicsBuilder.buildPhysics(world, mapData);
        Vector2 spawnPosition = getSpawnPosition(mapData);
        if (spawnPosition == null) {
            System.err.println("No player spawn found! Using default (0,0).");
            spawnPosition = new Vector2(0, 0);
        }
        GameStateManager.getInstance().updateSafePosition(spawnPosition.x, spawnPosition.y);
        knight = new Knight(world, spawnPosition.x * Knight.PPM, spawnPosition.y * Knight.PPM, 40f, 80f);
        knightController = new KnightController(knight);
        compositeContactListener = new CompositeContactListener();
        compositeContactListener.addListener(new KnightContactListener(knight));
        transitionManager = new TransitionManager(knight, mapManager, world);
        transitionManager.setWorldBodies(worldBodies);
        compositeContactListener.addListener(transitionManager);
        world.setContactListener(compositeContactListener);
        knightAtlas = new TextureAtlas(Gdx.files.internal("assets/knight.atlas"));
        knightAnimator = new KnightAnimator(knight, knightAtlas);
        camera.position.set(spawnPosition.x, spawnPosition.y, 0);
        camera.update();
    }
    private Vector2 getSpawnPosition(MapData mapData) {
        if (mapData == null || mapData.getSpawns().size == 0) {
            return null;
        }
        for (SpawnData spawn : mapData.getSpawns()) {
            if (spawn.getType() == SpawnType.PLAYER) {
                return spawn.getPosition().cpy().scl(1f / Knight.PPM);
            }
        }
        return mapData.getSpawns().first().getPosition().cpy().scl(1f / Knight.PPM);
    }
    private void update(float delta) {
        world.step(1 / 60f, 6, 2);
        transitionManager.update();
        knightController.update(delta);
        float mapWidth = mapManager.getMapWidth();
        float mapHeight = mapManager.getMapHeight();
        if (mapWidth == 0 || mapHeight == 0) {
            camera.position.set(knight.getBody().getPosition().x, knight.getBody().getPosition().y, 0);
        } else {
            float halfViewportWidth = viewport.getWorldWidth() / 2f;
            float halfViewportHeight = viewport.getWorldHeight() / 2f;
            float targetX = knight.getBody().getPosition().x;
            float targetY = knight.getBody().getPosition().y;
            float clampedX = Math.max(halfViewportWidth, Math.min(mapWidth - halfViewportWidth, targetX));
            float clampedY = Math.max(halfViewportHeight, Math.min(mapHeight - halfViewportHeight, targetY));
            camera.position.set(clampedX, clampedY, 0);
        }
        camera.update();
    }
    @Override
    public void render(float delta) {
        update(delta);
        Gdx.gl.glClearColor(0, 0, 0, 1);
        Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT);
        mapManager.render(camera);
        batch.setProjectionMatrix(camera.combined);
        batch.begin();
        TextureRegion frame = knightAnimator.getFrame(delta, knight.getCurrentState());
        if (frame != null) {
            float width = frame.getRegionWidth() / Knight.PPM;
            float height = frame.getRegionHeight() / Knight.PPM;
            float x = knight.getBody().getPosition().x - width / 2f;
            float y = knight.getBody().getPosition().y - height / 2f;
            batch.draw(frame, x, y, width, height);
        } else {
            System.err.println("Knight frame is null!");
        }
        batch.end();
        box2DDebugRenderer.render(world, camera.combined);
        super.render(delta);
    }
    @Override
    public void resize(int width, int height) {
        super.resize(width, height);
        viewport.update(width, height, true);
    }
    @Override
    public void dispose() {
        super.dispose();
        mapManager.dispose();
        world.dispose();
        box2DDebugRenderer.dispose();
        batch.dispose();
        knightAtlas.dispose();
    }
}

package ir.Ali.hollowknightme.lwjgl3;
import com.badlogic.gdx.Version;
import com.badlogic.gdx.backends.lwjgl3.Lwjgl3NativesLoader;
import org.lwjgl.system.JNI;
import org.lwjgl.system.linux.UNISTD;
import org.lwjgl.system.macosx.LibC;
import org.lwjgl.system.macosx.ObjCRuntime;
import java.io.File;
import java.lang.management.ManagementFactory;
import java.util.ArrayList;
import java.util.List;
import java.util.Locale;
public class StartupHelper {
	private StartupHelper() {}
	private static final String JVM_RESTARTED_ARG = "jvmIsRestarted";
	public static boolean isLinuxNvidia() {
		String[] drivers = new File("/proc/driver").list(
			(dir, path) -> path.toUpperCase(Locale.ROOT).contains("NVIDIA")
		);
		if (drivers == null) return false;
		return drivers.length > 0;
	}
	public static boolean startNewJvmIfRequired() {
		return startNewJvmIfRequired(true);
	}
	public static boolean startNewJvmIfRequired(boolean inheritIO) {
		String osName = System.getProperty("os.name").toLowerCase(Locale.ROOT);
		if (osName.contains("mac")) return startNewJvm0( true, inheritIO);
		if (osName.contains("windows")) {
			String programData = System.getenv("ProgramData");
			if (programData == null) programData = "C:\\Temp";
			String prevTmpDir = System.getProperty("java.io.tmpdir", programData);
			String prevUser = System.getProperty("user.name", "libGDX_User");
			System.setProperty("java.io.tmpdir", programData + "\\libGDX-temp");
			System.setProperty(
				"user.name",
				("User_" + prevUser.hashCode() + "_GDX" + Version.VERSION).replace('.', '_')
			);
			Lwjgl3NativesLoader.load();
			System.setProperty("java.io.tmpdir", prevTmpDir);
			System.setProperty("user.name", prevUser);
			return false;
		}
		return startNewJvm0( false, inheritIO);
	}
	private static final String MAC_JRE_ERR_MSG = "A Java installation could not be found. If you are distributing this app with a bundled JRE, be sure to set the '-XstartOnFirstThread' argument manually!";
	private static final String LINUX_JRE_ERR_MSG = "A Java installation could not be found. If you are distributing this app with a bundled JRE, be sure to set the environment variable '__GL_THREADED_OPTIMIZATIONS' to '0'!";
	private static final String CHILD_LOOP_ERR_MSG = "The current JVM process is a spawned child JVM process, but StartupHelper has attempted to spawn another child JVM process! This is a broken state, and should not normally happen! Your game may crash or not function properly!";
	public static boolean startNewJvm0(boolean isMac, boolean inheritIO) {
		long processID = getProcessID(isMac);
		if (!isMac) {
			if (!isLinuxNvidia()) return false;
			if ("0".equals(System.getenv("__GL_THREADED_OPTIMIZATIONS"))) return false;
		} else {
			if (!System.getProperty("org.graalvm.nativeimage.imagecode", "").isEmpty()) return false;
			long objcMsgSend = ObjCRuntime.getLibrary().getFunctionAddress("objc_msgSend");
			long nsThread = ObjCRuntime.objc_getClass("NSThread");
			long currentThread = JNI.invokePPP(nsThread, ObjCRuntime.sel_getUid("currentThread"), objcMsgSend);
			boolean isMainThread = JNI.invokePPZ(currentThread, ObjCRuntime.sel_getUid("isMainThread"), objcMsgSend);
			if (isMainThread) return false;
			if ("1".equals(System.getenv("JAVA_STARTED_ON_FIRST_THREAD_" + processID))) return false;
		}
		if ("true".equals(System.getProperty(JVM_RESTARTED_ARG))) {
			System.err.println(CHILD_LOOP_ERR_MSG);
			return false;
		}
		List<String> jvmArgs = new ArrayList<>();
		String javaExecPath = System.getProperty("java.home") + "/bin/java";
		if (!(new File(javaExecPath).exists())) {
			System.err.println(getJreErrMsg(isMac));
			return false;
		}
		jvmArgs.add(javaExecPath);
		if (isMac) jvmArgs.add("-XstartOnFirstThread");
		jvmArgs.add("-D" + JVM_RESTARTED_ARG + "=true");
		jvmArgs.addAll(ManagementFactory.getRuntimeMXBean().getInputArguments());
		jvmArgs.add("-cp");
		jvmArgs.add(System.getProperty("java.class.path"));
		String mainClass = System.getenv("JAVA_MAIN_CLASS_" + processID);
		if (mainClass == null) {
			StackTraceElement[] trace = Thread.currentThread().getStackTrace();
			if (trace.length > 0) mainClass = trace[trace.length - 1].getClassName();
			else {
				System.err.println("The main class could not be determined.");
				return false;
			}
		}
		jvmArgs.add(mainClass);
		try {
			ProcessBuilder processBuilder = new ProcessBuilder(jvmArgs);
			if (!isMac) processBuilder.environment().put("__GL_THREADED_OPTIMIZATIONS", "0");
			if (!inheritIO) processBuilder.start();
			else processBuilder.inheritIO().start().waitFor();
		} catch (Exception e) {
			System.err.println("There was a problem restarting the JVM.");
			e.printStackTrace();
		}
		return true;
	}
	private static String getJreErrMsg(boolean isMac) {
		if (isMac) return MAC_JRE_ERR_MSG;
		else return LINUX_JRE_ERR_MSG;
	}
	private static long getProcessID(boolean isMac) {
		if (isMac) return LibC.getpid();
		else return UNISTD.getpid();
	}
}
